File to DataURI (base64)

A cross-browser solution to read a local file and provide the base64 encoded content

Download this project as a .zip file Download this project as a tar.gz file

Example

For the browsers that don't support the FileReader API we'll load the Flash object:

var Flash = {
  getFileData: function(base64) {
    showResult(base64)
  }
};

function showResult(b) {
  $('#result').html(base64)
}

if (typeof FileReader !== "function")
  swfobject.embedSWF("dist/FileToDataURI.swf", "file-object", "100px", "100px", "10.0.0", "dist/expressInstall.swf", {}, {}, {});
else {
  // replace the <object> by a input file
  $('#file-object').replaceWith('<input type="file" id="file-object" value="Load a file" />').on('change', function(e) {
    var files = e.target.files,file;

    if (!files || files.length == 0) return;
    file = files[0];

    var fileReader = new FileReader();
    fileReader.onload = function (e) {
      showResult(e.target.result);
    };
    fileReader.readAsDataURL(file);
  });
}
